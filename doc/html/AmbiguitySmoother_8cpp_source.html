58  satTypeValueMap& AmbiguitySmoother::Process( const CommonTime& epoch,
59  satTypeValueMap& gData)
60  throw(ProcessingException)

63  try
64  {
65 
66  double ambValue(0.0);
67  int currentSize;
68 
69  SatIDSet satRejectedSet;
70 
71  // Loop through all satellites
72  satTypeValueMap::iterator it;
73  for (it = gData.begin(); it != gData.end(); ++it)
74  {
75 
76  try
77  {
78  // Try to extract the values
79  ambValue = (*it).second(smoothType);
80  }
81  catch(...)
82  {
83  // If some value is missing, then schedule this satellite
84  // for removal
85  satRejectedSet.insert( (*it).first );
86  continue;
87  }
88 
89  // First check if this satellite has previous arc information
90  if ( satArcMap.find( (*it).first ) == satArcMap.end() )
91  {
92  // If it doesn't have an entry, insert one
93  satArcMap[ (*it).first ] = 0.0;
94  }
95 
96  //>>> Start to smooth the ambiguity data
97 
98  // Then, check both if there is arc information, and if current
99  // arc number is different from arc number in storage (which
100  // means a cycle slip happened), reset the buffer for smoothing data.
101  if ( (*it).second.find(TypeID::satArc) != (*it).second.end() &&
102  (*it).second(TypeID::satArc) != satArcMap[ (*it).first ] )
103  {
104  // If different, update satellite arc in storage
105  satArcMap[ (*it).first ] = (*it).second(TypeID::satArc);
106 
107  // Reset the buffer
108  smoothingData[ (*it).first ].Buffer.clear();
109  smoothingData[ (*it).first ].Buffer.push_back(ambValue);
110 
111  // Now, the mean value equals with the current 'ambValue'
112  smoothingData[ (*it).first ].mean = ambValue ;
113 
114  // Set the currentSize
115  currentSize = 1;
116 
117  }
118  else
119  {
120  // Insert current value of smoothType to the end of buffer
121  smoothingData[ (*it).first ].Buffer.push_back( ambValue );
122 
123  // Get current window size
124  currentSize = smoothingData[ (*it).first ].Buffer.size();
125 
126  // Check if we have exceeded maximum window size
127  if(currentSize > windowSize)
128  {
129  // Get rid of oldest data, which is at the beginning of deque
130  smoothingData[ (*it).first ].Buffer.pop_front();
131  }
132 
133  // Update current window size
134  currentSize = smoothingData[ (*it).first ].Buffer.size();
135 
136  // Now, Let's compute the mean values and variance for the smoothingData
137  PowerSum errorStats;
138 
139  // Now, Let's insert all the data into errorStats.
140  for(int i = 0; i < currentSize; i++ )
141  {
142  errorStats.add( smoothingData[ (*it).first ].Buffer[i] );
143  }
144 
145  // Now, Let's compute the average value
146  smoothingData[ (*it).first ].mean = errorStats.average();
147  smoothingData[ (*it).first ].variance = errorStats.variance();
148 
149  } // End of smoothing data
150 
151 
152  // Insert the mean value into 'gdsMap'
153  // only change the values, when the 'mean' operator is realy done.
154  if( windowSize > 1 )
155  {
156  (*it).second[ smoothType ] = smoothingData[(*it).first].mean;
157  }
158 
159  } // End of 'for (it = gData.begin(); it != gData.end(); ++it)'
160 
161  // Remove satellites with missing data
162  gData.removeSatID(satRejectedSet);
163 
164  return gData;
165 
166  }
167  catch(Exception& u)
168  {
169  // Throw an exception if something unexpected happens

171  + StringUtils::asString( getIndex() ) + ":"
172  + u.what() );

174  GPSTK_THROW(e);
175 
176  }
