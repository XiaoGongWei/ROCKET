117  // Prepare set of current unknowns and list of current constraints
118  EdgeSet IndepAmbiguityDatum::prepareCurrentEdges( gnssDataMap& gdsMap )
119  {
120 
121  // Let's create current edge set
122  EdgeSet currentEdgeSet;
123 
124  // The ambiguity fixing method
125  ARRound ambRes;
126 
127  // Iterate through all items in the gnssDataMap
128  for( gnssDataMap::const_iterator it = gdsMap.begin();
129  it != gdsMap.end();
130  ++it )
131  {
132  // Then, iterate through corresponding 'sourceDataMap'
133  for( sourceDataMap::const_iterator sdmIter = (*it).second.begin();
134  sdmIter != (*it).second.end();
135  ++sdmIter )
136  {
137  // Finally, iterate through corresponding 'satTypeValueMap'
138  for( satTypeValueMap::const_iterator stvmIter =
139  (*sdmIter).second.begin();
140  stvmIter != (*sdmIter).second.end();
141  stvmIter++ )
142  {
143 
144  // New variable for the given 'ambType'
145  Variable var( ambType );
146  var.setSource( (*sdmIter).first );
147  var.setSatellite( (*stvmIter).first );
148 
149  double amb, ambVar, weight, elev;
150  int satArc;
151 
152  amb = ambMap[var];
153  ambVar = ambVarMap[ var ][ var ];
154 
155  // Weight for the edge
156  weight = 1.0/ambVar;
157 
158  // Get the arc for this edge
159  elev = (*stvmIter).second.getValue( TypeID::elevation );
160  satArc = (*stvmIter).second.getValue( TypeID::satArc );
161 
162  // Create a new edge
163  Edge edge( (*sdmIter).first, (*stvmIter).first, satArc, weight, elev );
164 
165  // New constraint is complete: Add it to 'currentEdgeSet'
166  currentEdgeSet.insert( edge );
167 
168  } // End of 'for( satTypeValueMap::const_iterator stvmIter = ...'
169 
170  } // End of 'for( sourceDataMap::const_iterator sdmIter = ...'
171 
172  } // End of 'for( gnssDataMap::const_iterator it = ...'
173 
174  // Return set of current edges
175  return currentEdgeSet;
176 
177  } // End of method 'IndepAmbiguityDatum::prepareCurrentEdges()'

181  // Prepare set of current unknowns and list of current constraints
182  EdgeSet IndepAmbiguityDatum::prepareCurrentIndepEdges()
183  {
184  // Current independent edge set
185  EdgeSet currentIndepEdgeSet;
186 
187  // Continuous independent edges
188  EdgeSet continuousIndepEdges;
189 
190  // Find the edge of the oldIndepEdges in currentEdges
197 
205 
213 
214  // Class of the Kruskal algorithm, which will generate independent edges
215  Kruskal kruskal(currentEdges);
216 
217  // Set initial independent edges
218  kruskal.setInitialEdges(continuousIndepEdges);
219 
220  // Create the Minimum Spanning Tree from the 'currentEdges'
221  currentIndepEdgeSet = kruskal.createMST();
222 
227  // Clear the independent ambiguity map firstly
228  IndepAmbMap.clear();
229 
230  // Visit each "Edge" in "currentIndepEdgeSet"
231  for( EdgeSet::const_iterator itEdge = currentIndepEdgeSet.begin();
232  itEdge != currentIndepEdgeSet.end();
233  ++itEdge)
234  {
235 
236  // We will work with a copy of current Variable
237  Variable var( ambType );
238 
239  // Set SourceID
240  var.setSource( (*itEdge).getSource() );
241 
242  // Set satellite
243  var.setSatellite( (*itEdge).getSatellite() );
244 
245 
246  // Cutting decision
247  double cutDec(1000.0);
248 
249  ARRound ambRes(1000,0.15,0.15);
250 
251  // Ambiguity value
252  double amb = ambMap[ var ];
253  double ambSig = std::sqrt( ambVarMap[var][var] );
254 
255  //
256  // To be modified, print out the independent ambiguities that
257  // can't be fixed with enough possibilities.
258  //
259  double decision = ambRes.getDecision(amb, ambSig);
260 
261  if(decision<cutDec)
262  {
263  cout << "float independent ambiguity are chosen!"
264  << StringUtils::asString(var) << endl;
265  }
266 
267  // Get the neareast integer ambiguity
268  double ambFixed = std::floor(amb+0.5);
269 
270  // Store the fixed ambiguity value into 'IndepAmbMap'
271  // Warning: we will insert this satellite into 'IndepAmbMap'
272  // directly whether it can be fixed or not, because
273  // this satellite is the independent ambiguity datum.
274  IndepAmbMap[ var ] = ambFixed;
275 
276  } // End of 'for( EdgeSet::const_iterator itEdge = ...'
277 
278  // Return set of current unknowns
279  return currentIndepEdgeSet;
280 
281  } // End of method 'IndepAmbiguityDatum::prepareCurrentIndepEdges()'
