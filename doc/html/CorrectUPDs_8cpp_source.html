49  /* Returns a satTypeValueMap object, adding the new data generated
50  * when calling this object.
51  *
52  * @param time Epoch corresponding to the data.
53  * @param gData Data object holding the data.
54  */
55  satTypeValueMap& CorrectUPDs::Process( const CommonTime& time,
56  satTypeValueMap& gData )
57  throw(ProcessingException)
58  {
59  try
60  {
61  SatIDSet satRejectedSet;
62 
63  // Loop through all the satellites
64  for (satTypeValueMap::iterator stv = gData.begin();
65  stv != gData.end();
66  ++stv)
67  {
68 
69  try
70  {
71  // Get satellite upds from 'RinexUPDStore'
72  satUPD = (*pUPDStore).getValue( (*stv).first, time );
73 
74  }
75  catch(InvalidRequest& e)
76  {
77  // If some problem appears, then schedule this satellite
78  // for removal
79  satRejectedSet.insert( (*stv).first );
80 
81  continue; // Skip this SV if problems arise
82 
83  }
84 
85  // First, correct from cycles to meters.
86  // Warning:
87  // you should check the unit of the UPDs in the input file.
88  // if it is cycles, then convert it to meters firstly,
89  // or if it is meters, you can use it directly.
90  //
91  // Here, change the unit to meters firstly.
92  // 2015/09/15
93 // satUPD.updSatMW = satUPD.updSatMW*0.861918400322;
94 // satUPD.updSatLC = satUPD.updSatLC*0.106953378142;
95 
96  // Now, correct the Melboune-Wubbena combination observables
97  (*stv).second[TypeID::updSatMW] = satUPD.updSatMW;
98  (*stv).second[TypeID::updSatWL] = satUPD.updSatMW;
99  (*stv).second[TypeID::updSatLC] = satUPD.updSatLC;
100 
101  double f1 = 1575.42e6;
102  double f2 = 1227.60e6;
103 
104  double updSatWL, updSatLC;
105  double updSatL1, updSatL2;
106 
107  // Change unit from meter to cycles
108  updSatWL = satUPD.updSatMW/0.861918400322;
109  updSatLC = satUPD.updSatLC/0.106953378142;
110 
111  // Get updSatL1/updSatL2, unit:cycles
112  updSatL1 = updSatLC - f2/(f1-f2)*updSatWL;
113  updSatL2 = updSatLC - f1/(f1-f2)*updSatWL;
114 
115  // Convert from cycles to meters
116  updSatL1 = updSatL1 * 0.190293672798;
117  updSatL2 = updSatL2 * 0.244210213425;
118 
119  (*stv).second[TypeID::updSatL1] = updSatL1;
120  (*stv).second[TypeID::updSatL2] = updSatL2;
121 
122 
123  } // End of 'for (stv = gData.begin(); stv != gData.end(); ++stv)'
124 
125  // Remove satellites with missing data
126  gData.removeSatID(satRejectedSet);
127 
128  // Return
129  return gData;
130 
131  }
132  catch(Exception& u)
133  {
134 
135  // Throw an exception if something unexpected happens
136  ProcessingException e( getClassName() + ":"
137  + StringUtils::asString( getIndex() ) + ":"
138  + u.what() );
139 
140  GPSTK_THROW(e);
141 
142  }
143 
144  } // End of method 'CorrectUPDs::Process()'
